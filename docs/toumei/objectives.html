10class Pipeline(Objective):
11    """
12    This is a wrapper class for objective generation in a pipeline based work-flow.
13    It takes a generator and a tree like objective function definition consisting of modules, which is executed
14    recursively in the forward function.
15    """
16    def __init__(self, img_generator: ImageGenerator, obj_func: Atom):
17        super(Pipeline, self).__init__()
18
19        self.model = None
20
21        # the image generator and the objective function
22        self.img_generator = img_generator
23        self.obj_func = obj_func
24
25    def attach(self, model: nn.Module):
26        """
27        Attach the modules to the model.
28        :param model: the model
29        """
30        if self.model is not None:
31            self.detach()
32
33        self.model = model
34
35        # freeze the model
36        freeze_model(model)
37
38        # call attach on the root
39        self.root.attach(model)
40
41    def detach(self):
42        """
43        Detach the modules from the current model
44        :return: nothing
45        """
46
47        # check if attached
48        if self.model is None:
49            print("Cannot detach the current objective, since it was not detached in the first place.")
50            return
51
52        # unfreeze the model
53        unfreeze_model(self.model)
54
55        # call detach on the root
56        self.root.detach()
57
58        # reset the current model
59        self.model = None
60
61    @property
62    def root(self) -> Atom:
63        """
64        Returns the root node of the objective function tree
65        :return: the root node
66        """
67        return self.obj_func
68
69    @property
70    def generator(self) -> ImageGenerator:
71        """
72        Returns the image generator
73        :return:
74        """
75        return self.img_generator
76
77    def forward(self) -> torch.Tensor:
78        """
79        The forward function for the objective.
80        This calls the root, which induces recursive calls for each child of the module
81        :return: a tensor for optimization
82        """
83        return self.root()
